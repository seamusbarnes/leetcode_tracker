class Solution:
    def longestIdealString(self, s: str, k: int) -> int:

        '''
        subsequence problems lend themselves to dynamic programming (DP)
        recursive solution: each iteration, you split down a decision tree
        '''

        # recursive solution (memory limit exceeded)
        def recursive_solution():
            def helper(i, prev):
                if i == len(s):
                    return 0

                # cache
                if (i, prev) in cache:
                    return cache[(i, prev)]

                # skip s[i]
                res = helper(i + 1, prev)

                # include s[i]
                if prev == "" or abs(ord(s[i]) - ord(prev)) <= k:
                    res = max(res, helper(i + 1, s[i]) + 1)


                cache[(i, prev)] = res
                return res

            cache = {}

            return helper(0, "")

        # return recursive_solution()

        # dynamic programming solution
        def dp_solution():
            dp = [0] * 26

            for c in s:
                curr = ord(c) - ord('a')
                longest = 1
                for prev in range(26):
                    if abs(curr - prev) <= k:
                        longest = max(longest, 1 + dp[prev])
                dp[curr] = longest


            return max(dp)

            return res

        return dp_solution()
